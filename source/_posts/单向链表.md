### 单向链表  
- 链表的一种最简单形式,每一个节点包含两个域,一个信息域(元素域)和一个链接域,链接指向下一个节点,最后一个节点的链接指向一个空值(None).  
- 单向循环链表则是在单向链表的基础上,将最后一个节点的链接指向链表的头节点.  
- 复杂一点的链表为双向链表,每个节点有两个链接:一个指向前一个节点,若此节点为第一个节点时,指向空值,而另一个指向下一个节点,当此节点为最后一个节点时,指向空值.

下面给出单向链表的实现:


```python
class SingleNode(object):
    '''单链表的结点'''
    def __init__(self, item):
        self.item = item   #item存放数据元素
        self.next = None   #next是下一个节点的标识     
```


```python
class SingleLinkList(object):
    '''单链表'''
    def __init__(self):
        self._head = None
    
    def is_empty(self):
        '''判断链表是否为空'''
        return self._head == None
    
    def length(self):
        '''链表的长度'''
        cur = self._head  #cur初始时指向头结点
        count = 0
        while cur != None:  #尾节点指向None,当未到达尾部时
            count += 1
            cur = cur.next  #将cur后移一个节点
        return count
    
    def travel(self):
        '''遍历链表'''
        cur = self._head
        while cur != None:
            print(cur.item)
            cur = cur.next
        print("")
    
    def add(self, item):
        '''头部添加元素'''
        node = SingleNode(item) #创建一个保存Node的节点
        node.next = self._head
        self._head = node
        
    def append(self, item):
        '''尾部添加元素'''
        node = SingleNode(item)
        if self.is_empty():    #先判断链表是否为空
            self._head = node
        else:
            cur = self._head
            while cur.next != None:
                cur = cur.next
            cur.next = node
    
    def insert(self, pos, item):
        '''指定位置添加元素'''
        if pos <= 0:          #若指定位置在第一个元素之前,则执行头部插入
            self.add(item)
        elif pos > (self.length() - 1):      #若指定位置超过链表的尾部,执行尾部插入 
            self.append(item)
        else:
            node = SingleNode(item)
            count = 0
            pre = self._head
            while count < (pos - 1):
                count += 1
                pre = pre.next
            node.next = pre.next
            pre.next = node
    def remove(self, item):
        '''删除节点'''
        cur = self._head
        pre = None
        while cur != None:
            if cur.item == item:
                if not pre:       #第一个就是要删除的item
                    self._head = cur.next
                else:
                    pre.next = cur.next       #将删除位置前一个节点的next指向删除位置后一个节点
                break
            else:     #继续往后移动节点
                pre = cur
                cur = cur.next
                
    def search(self, item):
        '''链表查找节点是否存在,并返回True或者False'''
        cur = self._head
        while cur != None:
            if cur.item == item:
                return True
            cur = cur.next
        return False
```
